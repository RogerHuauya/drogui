\section{Ecuaciones de la dinamica}
\begin{gather*}
	\\
	\ddot{x} = 0.29*U_4*(sin(roll)*sin(yaw) + cos(roll)*cos(yaw)*sin(pitch))\\
	\ddot{y}= -0.29*U_4*(cos(yaw)*sin(roll) - 1.0*cos(roll)*sin(pitch)*sin(yaw))\\
	\\
\end{gather*}
\section{Espacio estados final}
